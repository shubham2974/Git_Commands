//To check Status
git status

//To Add file to staging stage
git add <file name. ext>

//To Skip Staging area
git commit -am "Message"

//To commit files in staging area with message
git commit -m "Message..."

//To delete file from staging
git rm --cached <File Name.ext>

//To create a branch
git branch <Branch Name>

//To work on branch
git checkout <Branch Name>

//To Merge the Branches, Come to master branch first
git merge <Branch Name>
